</div>

<PinyinPoem 
  title="龟虽寿" 
  author="曹操"
  content={[

    "老骥伏枥，志在千里。",
    "烈士暮年，壮心不已。",
    "盈缩之期，不但在天；",

    "幸甚至哉，歌以咏志。"
  ]}
  pinyinData={[
    ["shén guī suī shòu, yóu yǒu jìng shí."],

    ["yíng suō zhī qī, bù dàn zài tiān;"],
    ["yǎng yí zhī fú, kě dé yǒng nián."],
